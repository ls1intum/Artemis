import { definePreset } from '@primeuix/themes';
import Aura from '@primeuix/themes/aura';
 
/**
 * This overrides the PrimeNG Aura theme to use Artemis theme colors.
 * Instead of hardcoding colors, we reference CSS custom properties that are
 * automatically generated from SCSS variables in _default-variables.scss.
 * This ensures a single source of truth - when you update a color in the SCSS
 * theme file, PrimeNG components will automatically use the updated color.
 *
 * The CSS variables are generated by the _generate-css-vars.scss mixin.

    semantic: {
        primary: {
